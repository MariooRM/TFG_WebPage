/**
 * This is the store used for managing the user's games stats.
 * @module useStatsStore
 */
import { defineStore } from "pinia";
import { getFirestore, collection, onSnapshot } from "firebase/firestore";

/**
 * @typedef {Object} State
 * @property {number} totalGames - The total number of games.
 * @property {Object} gamesData - An object containing the user's games data.
 * @property {?function} suscription - The current Firestore subscription.
 */

/**
 * @typedef {Object} Actions
 * @property {function(string):void} fetchUserGamesDocs - Fetch the user's games data from Firestore.
 * @property {function():void} clearData - Clear data when logging out.
 */

export const useStatsStore = defineStore({
  id: 'stats',
  /**
   * @returns {State} The state object.
   */
  state: () => ({
    /** 
     * The total number of games.
     * @type {number}
     */
    totalGames: localStorage.getItem('totalGames') || 0,

    /**
     * An object containing the user's games data.
     * @type {Object}
     */
    gamesData: JSON.parse(localStorage.getItem('gamesData')) || {}, 

    /**
     * The current Firestore subscription.
     * @type {?function}
     */
    suscription: null, 
  }),

  actions: /** @type {Actions} */ ({
    /**
     * Fetch the user's games data from Firestore.
     * @param {string} userUID - The UID of the user.
     */
    fetchUserGamesDocs(userUID) {
      const db = getFirestore();
      const gamesRef = collection(db, `users/${userUID}/games`);

      // Manage previous suscription
      if (this.suscription) {
        this.suscription();
        this.suscription = null;
      }

      // Configure new suscription
      this.suscription = onSnapshot(gamesRef, (snapshot) => {
        this.totalGames = snapshot.size;
        localStorage.setItem('totalGames', this.totalGames);
        
        this.gamesData = snapshot.docs.reduce((acc, doc) => {
          acc[doc.id] = doc.data();
          return acc;
        }, {});
        
        localStorage.setItem('gamesData', JSON.stringify(this.gamesData));
      }, (error) => {
        console.error("Error fetching games data: ", error);
      });
    },

    /**
     * Clear data when logging out.
     */
    clearData() {
      localStorage.removeItem('totalGames');
      localStorage.removeItem('gamesData');
      this.totalGames = 0;
      this.gamesData = {};

      // Desuscribe from Firebase
      if (this.suscription) {
        this.suscription();
        this.suscription = null;
      }
    }
  })
});
